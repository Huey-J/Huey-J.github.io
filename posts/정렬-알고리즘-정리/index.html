def bubbleSort(x):
  length = len(x)-1
  for i in range(length):
    for j in range(length-i):
      if x[j] > x[j+1]:		# ì•ì—ê°€ ë” í¬ë©´ Swap
        x[j], x[j+1] = x[j+1], x[j]
  return x

def selectionSort(x):
  length = len(x)
  for i in range(length-1):
